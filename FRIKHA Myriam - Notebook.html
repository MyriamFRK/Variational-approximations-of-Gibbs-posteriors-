# Variational lower bound definition for F2
def variational_lower_bound(m, sigma, X, y, lambda_param, vartheta):
    n, d = X.shape
    lower_bound = 0
    for i in range(n):
        xi = X[i, :]
        yi = y[i]
        epsilon = 1e-8
        denominator = np.sqrt(np.dot(xi, sigma**2 * xi.T) + epsilon)
        argument = -(yi * np.dot(xi, m)) / denominator
        lower_bound += norm.cdf(argument)
    regularization_m = np.sum(m**2) / (2 * vartheta)
    regularization_sigma = 0.5 * np.sum(np.log(sigma**2) - (sigma**2 / vartheta))
    lower_bound = -lambda_param * lower_bound / n - regularization_m + regularization_sigma
    return -lower_bound

# Optimization function for variational parameters
def optimize_variational_parameters(X, y, lambda_param=1.0, vartheta=1.0):
    n, d = X.shape
    initial_m = np.zeros(d)
    initial_log_sigma = np.zeros(d)
    def objective(params):
        m, log_sigma = params[:d], params[d:]
        sigma = np.exp(log_sigma)
        return variational_lower_bound(m, sigma, X, y, lambda_param, vartheta)
    result = minimize(objective, np.concatenate([initial_m, initial_log_sigma]), method='L-BFGS-B')
    optimized_m = result.x[:d]
    optimized_sigma = np.exp(result.x[d:])
    return optimized_m, optimized_sigma

import numpy as np
from scipy.stats import multivariate_normal
from scipy.optimize import minimize

# Define the variational lower bound for F3, considering a full covariance matrix
def variational_lower_bound_F3(m, L, X, y, lambda_param, vartheta):
    n, d = X.shape
    lower_bound = 0
    # Reconstruct the covariance matrix from its Cholesky decomposition to ensure it's positive definite
    Sigma = np.dot(L, L.T) + np.eye(d) * 1e-3  # Add a small value for numerical stability

    epsilon = 1e-10  # Small constant to avoid log(0)
    
    # Compute the term for each data point
    for i in range(n):
        xi = X[i, :]
        yi = y[i]
        prob_density = multivariate_normal.pdf(xi, mean=m, cov=Sigma, allow_singular=True)
        # Add epsilon to prob_density inside the log to prevent log(0)
        lower_bound += np.log(prob_density + epsilon) if yi == 1 else np.log(1 - prob_density + epsilon)
    
    # Regularization terms for mean and covariance
    regularization_m = np.sum(m**2) / (2 * vartheta)
    regularization_Sigma = np.sum(np.log(np.diag(Sigma))) - np.sum(Sigma) / vartheta
    
    # Combine all terms
    lower_bound = lambda_param * lower_bound / n - regularization_m + regularization_Sigma
    
    return -lower_bound  # Negative because we minimize in optimization

# Example optimization of the variational parameters (m, L) using L-BFGS-B
def optimize_variational_parameters_F3(X, y, lambda_param=1.0, vartheta=1.0):
    n, d = X.shape
    
    # Initial parameters: mean vector (m) and Cholesky decomposition of covariance matrix (L)
    initial_m = np.zeros(d)
    initial_L = np.linalg.cholesky(np.eye(d))  # Initial L as Cholesky decomposition of identity matrix
    
    # Flatten L to use in optimization
    initial_params = np.concatenate([initial_m, initial_L[np.tril_indices(d)]])
    
    # Objective function to minimize
    def objective(params):
        m = params[:d]
        L_flat = params[d:]
        # Reconstruct the lower triangular part of L from the flattened version
        L = np.zeros((d, d))
        L[np.tril_indices(d)] = L_flat
        return variational_lower_bound_F3(m, L, X, y, lambda_param, vartheta)
    
    # Minimize the negative variational lower bound
    result = minimize(objective, initial_params, method='L-BFGS-B')
    
    optimized_m = result.x[:d]
    optimized_L_flat = result.x[d:]
    optimized_L = np.zeros((d, d))
    optimized_L[np.tril_indices(d)] = optimized_L_flat
    
    return optimized_m, optimized_L

def predict_F3(X_test, m_opt, L_opt):
    """
    Predict function for F3 using the optimized mean vector and Cholesky factor
    of the covariance matrix. This example predicts based on the density; a more complex
    or appropriate decision rule might be needed based on application specifics.
    """
    Sigma = np.dot(L_opt, L_opt.T) + np.eye(L_opt.shape[0]) * 1e-3  # Reconstruct full covariance matrix

    # Prediction based on whether a sample's density under the model is above a certain threshold
    pred_labels = [1 if multivariate_normal.pdf(x, mean=m_opt, cov=Sigma) > 0.5 else 0 for x in X_test]
    return np.array(pred_labels)

def evaluate_F3_model(X_train, y_train, X_test, y_test, lambda_param=1.0, vartheta=1.0):
    """
    Evaluates the F3 model by optimizing variational parameters, making predictions on test data,
    and calculating the misclassification rate.
    """
    m_opt, L_opt = optimize_variational_parameters_F3(X_train, y_train, lambda_param, vartheta)
    y_pred = predict_F3(X_test, m_opt, L_opt)
    mis_rate = calculate_misclassification_rate(y_test, y_pred)
    return mis_rate , y_pred 

def calculate_misclassification_rate(y_true, y_pred):
    """
    Calculate the misclassification rate given true labels and predicted labels.
    """
    misclassification_rate = np.mean(y_true != y_pred) * 100  # Percentage of incorrect predictions
    return misclassification_rate
